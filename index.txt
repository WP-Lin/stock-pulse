// Convert user input → TradingView symbol format
function toTVSymbol(raw) {
  const s = raw.trim().toUpperCase();
  if (s.includes(':')) return s;              
  if (s.endsWith('.TW'))  return 'TWSE:' + s.replace('.TW','');
  if (s.endsWith('.TWO')) return 'TPEX:' + s.replace('.TWO','');
  if (s.endsWith('.HK'))  return 'HKEX:' + s.replace('.HK','');
  if (s.endsWith('.T'))   return 'TSE:' + s.replace('.T','');    
  return s;
}

// Human-readable display symbol
function displaySym(tvSym) {
  if (!tvSym) return '';
  if (tvSym.startsWith('TWSE:'))  return tvSym.replace('TWSE:','') + '.TW';
  if (tvSym.startsWith('TPEX:'))  return tvSym.replace('TPEX:','') + '.TWO';
  if (tvSym.startsWith('TSE:'))   return tvSym.replace('TSE:','') + '.TW'; // for backwards compatibility
  if (tvSym.startsWith('HKEX:')) return tvSym.replace('HKEX:','') + '.HK';
  return tvSym.includes(':') ? tvSym.split(':')[1] : tvSym;
}

      updateSymBar(symbol);
    },
  });
  
  // 即使圖表還沒準備好，先更新頂部資訊列
  updateSymBar(symbol);
}

function selectSym(tvSym) {
  curSym = tvSym;
  renderWatchlist();
  buildWidget(tvSym);

function addFromSearch() {
  const inp = document.getElementById('searchInput');
  const raw = inp.value.trim();
  if (!raw) return;
  const tvSym = toTVSymbol(raw);
  if (!watchlist.includes(tvSym)) {
    watchlist.push(tvSym);
    saveWL();
    toast('✓ 已加入 ' + displaySym(tvSym));
  }
  inp.value = '';
  selectSym(tvSym);
}

function rmSym(tvSym, ev) {
  ev.stopPropagation();
  watchlist = watchlist.filter(s => s !== tvSym);
  saveWL();
  if (curSym === tvSym && watchlist.length) selectSym(watchlist[0]);
  renderWatchlist();
  toast('已移除 ' + displaySym(tvSym), 'ng');
}

  if (!confirm('確定清空自選清單？')) return;
  watchlist = []; saveWL(); renderWatchlist();
  document.getElementById('tv-dynamic-container').innerHTML = `
